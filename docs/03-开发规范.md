# GoldPilot 开发规范

## 一、代码规范

### 1.1 命名规范

#### 文件命名
```
组件文件：PascalCase（如：PriceCard.tsx）
工具文件：camelCase（如：format.ts）
常量文件：camelCase（如：constants.ts）
类型文件：camelCase（如：types.ts）
```

#### 变量命名
```typescript
// 常量：UPPER_SNAKE_CASE
const MAX_SIGNAL_COUNT = 100;
const DEFAULT_TIMEOUT = 5000;

// 变量/函数：camelCase
let currentPrice = 0;
function calculateSignals() {}

// 类/接口/类型：PascalCase
class SignalService {}
interface CandleData {}
type SignalDirection = 'long' | 'short';

// 私有变量：前缀下划线
class MyClass {
  private _internalValue: number;
}
```

### 1.2 TypeScript规范

#### 类型定义
```typescript
// ✅ 推荐：使用interface定义对象类型
interface Signal {
  id: string;
  direction: 'long' | 'short';
  entryPrice: number;
  exitPrice?: number;
}

// ✅ 推荐：使用type定义联合类型
type SignalDirection = 'long' | 'short';
type SignalStatus = 'pending' | 'profit' | 'loss';

// ✅ 推荐：导出类型
export type { Signal, SignalDirection };
```

#### 类型注解
```typescript
// ✅ 推荐：函数参数和返回值都要有类型注解
function calculateEMA(data: number[], period: number): number {
  // ...
}

// ✅ 推荐：解构时注明类型
const { price, change }: { price: number; change: number } = getData();

// ❌ 避免：使用any
function process(data: any) { }
```

#### 泛型使用
```typescript
// ✅ 推荐：使用有意义的泛型名
function identity<T>(value: T): T {
  return value;
}

// ✅ 推荐：泛型约束
interface Lengthwise {
  length: number;
}
function logLength<T extends Lengthwise>(arg: T): void {
  console.log(arg.length);
}
```

### 1.3 React规范

#### 组件定义
```typescript
// ✅ 推荐：函数组件 + Hooks
interface PriceCardProps {
  price: number;
  change: number;
}

export function PriceCard({ price, change }: PriceCardProps) {
  return <div>{price}</div>;
}

// ❌ 避免：类组件（除非必要）
class PriceCard extends Component {
  // ...
}
```

#### Hooks使用
```typescript
// ✅ 推荐：自定义Hook
function useWebSocket(url: string) {
  const [socket, setSocket] = useState<Socket | null>(null);

  useEffect(() => {
    const socket = io(url);
    setSocket(socket);
    return () => socket.close();
  }, [url]);

  return socket;
}

// ✅ 推荐：Hook命名以use开头
function useSignals() { }
function useAccountData() { }

// ❌ 避免：在条件/循环中使用Hook
if (condition) {
  useEffect(() => { }); // 错误！
}
```

#### 状态管理
```typescript
// ✅ 推荐：简单状态使用useState
const [price, setPrice] = useState<number>(0);

// ✅ 推荐：复杂对象使用useReducer
const [state, dispatch] = useReducer(signalReducer, initialState);

// ✅ 推荐：全局状态使用Context
const AppContext = createContext<AppContextType | null>(null);

// ✅ 推荐：服务器状态使用React Query/SWR
const { data, error } = useSWR('/api/signals', fetcher);
```

#### TailwindCSS使用

// ✅ 推荐：使用Tailwind类名

## 三、API规范

### 3.1 REST API设计

#### 命名规范
```
GET    /api/signals        # 获取列表
GET    /api/signals/:id    # 获取单个
POST   /api/signals        # 创建
PUT    /api/signals/:id    # 更新
DELETE /api/signals/:id    # 删除
```

#### 响应格式
```typescript
// ✅ 成功响应
{
  "success": true,
  "data": {
    "signals": [...]
  }
}

// ❌ 错误响应
{
  "success": false,
  "error": {
    "code": "INVALID_PARAMS",
    "message": "参数错误"
  }
}
```

### 3.2 WebSocket规范

#### 事件命名
```typescript
// 客户端 → 服务端：使用动词
emit: 'subscribe:price'
emit: 'unsubscribe:price'
emit: 'request:signals'

// 服务端 → 客户端：使用名词
on: 'price'
on: 'signal'
on: 'error'
```

#### 数据格式
```typescript
// 统一数据包格式
interface DataPacket<T> {
  type: string;
  data: T;
  timestamp: number;
}

// 示例
socket.emit('price', {
  type: 'price',
  data: { symbol: 'XAU/USD', price: 2388.50 },
  timestamp: Date.now()
});
```
